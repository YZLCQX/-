CREATE TABLE web_user_info_table
(
id int AUTO_INCREMENT primary key ,
name char(16) NOT NULL,
password char(40) NOT NULL,
Creation_time char(24) NOT NULL,
user_imag  char(128)
);


CREATE TABLE web_cookie_table
(
id int AUTO_INCREMENT primary key,
cookie char(40) NOT NULL,
time int NOT NULL
);


ALTER TABLE web_cookie_table ADD user_name char(16) NOT NULL AFTER cookie;
